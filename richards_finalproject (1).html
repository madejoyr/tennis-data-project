def get_wiki_data(url):
    page = requests.get(url)


    soup = BeautifulSoup(page.content, 'html.parser')

    table = soup.find_all('table', class_="sortable wikitable")[0]

    rows = table.findAll('tr')


    all_data = []
    all_title_data = []


    for row in rows:
        links = row.findAll('a')
        all_data.append(links) 


    for row_links in all_data:
        title_data = []
        if len(row_links) > 0:
            for link in row_links:
                title = link.get('title')
                title_data.append(title)
        all_title_data.append(title_data)

        
    etc = []

    for lst in all_title_data:
        if len(lst) > 0:
            etc.append(lst[0])


    dx = pd.DataFrame()
    dx['Data'] = etc
    
    wl = table.findAll(('td'), {'style':re.compile("background:#98fb98|background:#ffa07a")})

    wl_list = []

    for row in wl:
        row = row.renderContents()
        encoding = 'utf-8'
        row = row.decode(encoding)
        wl_list.append(row)
    
    df = pd.DataFrame()
    df['result'] = wl_list
          
    m = pd.concat([df, dx], join="outer", axis = 1)
    print(m)
    
    m['Data']= m['Data'].str.replace(r'  Women\'s Singles','')  
    m['Data']= m['Data'].str.replace(r' \(tennis\)','') 
    m[['year', 'tournament']] = m["Data"].str.extract('(?P<year>\d{4}) (?P<tornament>.+)') 
    m = m[["result", "year", "tournament"]]

    return(m)

Scraping the data from twitter was challenging because of the twitter constraints. You can only scrape 200 tweets at a time, only make 15 calls in 15 minutes, and only get data on the most recent 3,240 tweets. The challenge was getting the next 200 tweets without duplicating the data. After hours of trial and error, I decided to use GitHub user yanofsky's code on twitter as a guide (https://gist.github.com/yanofsky/5436496). The code originally wrote the data to a CSV file, so I took that part out and wrote it to a pandas data frame instead. The function takes in the twitter user-name, or handle, and returns the data frame of twitter data.

def get_all_tweets(screen_name):
    #Twitter only allows access to a users most recent 3240 tweets
    
    #authorize twitter, initialize tweepy
    auth = tweepy.OAuthHandler(consumer_key, consumer_secret)
    auth.set_access_token(access_key, access_secret)
    api = tweepy.API(auth)
    
    #initialize a list to hold all the tweepy Tweets
    alltweets = []
    all_tweets = []
    
    #make initial request for most recent tweets (200 is the maximum allowed count)
    new_tweets = api.user_timeline(screen_name = screen_name,count=200)
    #tweets = [i.AsDict() for i in new_tweets]
    
    #save most recent tweets
    alltweets.extend(new_tweets)
    
    #save the id of the oldest tweet less one
    oldest = alltweets[-1].id - 1
    
    #keep grabbing tweets until there are no tweets left to grab
    while len(new_tweets) > 0:
        
        #all subsiquent requests use the max_id param to prevent duplicates
        new_tweets = api.user_timeline(screen_name = screen_name,count=200,include_rts=False,max_id=oldest)
       
        #save most recent tweets
        alltweets.extend(new_tweets)
        
        #update the id of the oldest tweet less one
        oldest = alltweets[-1].id - 1

        for tweet in alltweets:
            time = tweet.created_at
            retweet = tweet.retweet_count  
            favorites = tweet.favorite_count

            twitter_data = {'time':time,'retweet':retweet, 'favorites':favorites}
            all_tweets.append(twitter_data)
   
    df = pd.DataFrame(all_tweets)
    df['time']=pd.to_datetime(df['time'])
    df = df.set_index('time')

    return(df)
